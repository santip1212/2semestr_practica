cmake_minimum_required(VERSION 3.10)

project(ImageCompressor)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

set(SOURCES
    main.cpp
    ImageCompressorApp.cpp
)
 
set(HEADERS
    ImageCompressorApp.h
)

add_executable(ImageCompressor ${SOURCES} ${HEADERS})

qt6_wrap_cpp(MOC_HEADERS ${HEADERS})

target_sources(ImageCompressor PRIVATE ${MOC_HEADERS})

target_link_libraries(ImageCompressor 
    Qt6::Widgets
)
